This is the main include file necessary in order to use the W3C Reference Library. Currently the file only consists of well-known Library include files, but the goal is to define a nice API in this file, so that all external functionality can be reached through this one. 
#ifndef WWWLIB_H
#define WWWLIB_H

#ifdef __cplusplus
extern "C" { 
#endif

System dependencies

#include "tcp.h"

General Utilities

#include "HTUtils.h"			/* Macros and other stuff */
#include "HTString.h"			/* Generic string utilities */
#include "HTList.h"			/* Linked lists */
#include "HTChunk.h"			/* Dynamic arrays */
#include "HTAtom.h"			/* Pointer referenced strings */
#include "HTParse.h"			/* Parse URLs */

Core Modules

/* Format Management, and Access functions for loading a URL etc. */
#include "HTMethod.h"			/* Request methods like PUT, GET */
#include "HTFormat.h"			/* Stream Stack and content neg. */
#include "HTReq.h"			/* public part of request manager */
#include "HTAnchor.h"			/* Anchor class Definition */
#include "HTEvntrg.h"
#include "HTError.h"
#include "HTProt.h"

/* Access Authentication */
#include "HTAAUtil.h"
#include "HTUU.h"

/* Cache manager */
#include "HTCache.h"

/* Presenters, converters and other streams */
#include "HTMLGen.h"			/* Generating HTML */
#include "HTWriter.h"			/* Write to a unix file descriptor */
#include "HTFWrite.h"			/* Write to an ANSI file descriptor */
#include "HTMIME.h"			/* HTTP/MIME parsing */
#include "HTNetTxt.h"			/* Convert from CRLF to \n */

/* Directory Listings */
#include "HTIcons.h"
#include "HTDirBrw.h"
#include "HTDescpt.h"

/* Application specific modules */
#include "HTInit.h"			/* Initialization functions */
#include "HTLog.h"
#include "HTAccess.h"			/* Document access network code */
#include "HTHist.h"			/* Navigational aids */

/* For TCP relevant information (mail address etc.) */
#include "HTTCP.h"

/* Management of Rule Files for configuring the application */
#include "HTRules.h"
#include "HTProxy.h"
#include "HTBind.h"			/* Binding to file suffixes */

/* Event loop for multithreaded functionality */

/* Messages and User prompts */
#include "HTAlert.h"

#ifdef __cplusplus
} /* end extern C definitions */
#endif

#endif
End of WWWLib API definition
